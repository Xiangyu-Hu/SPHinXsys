36  // Compute the polar decomposition of the input matrix.
37  //
38  // This method decomposes the input matrix into one orthonormal (rotation) matrix
39  // and a remaining positive semi-definite (scale) matrix.
40  //
41  // Matrices are represented as 3x3 matrices in column-major representation.
42  //
43  // A [in] : Matrix to decompose.
44  // Q [out] : Rotation part of the polar decomposition.
45  // H [out] : Symmetric matrix represention the scale part of the polar decomposition.
46  template <typename TReal>
47  void polar_decomposition(TReal* Q, TReal* H, const TReal* A)
48  {
49  // Convert parameters for the algorithm implementation.
50  typedef detail::matrix<TReal, 3, 3> TMatrix;
51 
52  // Depending on the implementation, this part might involve a copy,
53  // but here the memory layout requirement is for matrices to be 3x3,
54  // column-major, so we can just cast directly.
55  TMatrix* matrixQ = reinterpret_cast<TMatrix*>(Q);
56  TMatrix* matrixH = reinterpret_cast<TMatrix*>(H);
57  const TMatrix* matrixA = reinterpret_cast<const TMatrix*>(A);
58 
59  detail::run_algorithm_3_5(*matrixQ, *matrixH, *matrixA);
60  }
