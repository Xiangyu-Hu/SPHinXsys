30  //Allocate and deallocate 3d array
31  //-------------------------------------------------------------------------------------------------
32  template<class T>
33  void Allocate3dArray(T*** &matrix, Vec3u res)
34  {
35  matrix = new T**[res[0]];
36  for (size_t i = 0; i < res[0]; i++) {
37  matrix[i] = new T*[res[1]];
38  for (size_t j = 0; j < res[1]; j++) {
39  matrix[i][j] = new T[res[2]];
40  }
41  }
42  }
43 
44  template<class T>
45  void Delete3dArray(T*** matrix, Vec3u res)
46  {
47  for (size_t i = 0; i < res[0]; i++) {
48  for (size_t j = 0; j < res[1]; j++) {
49  delete[] matrix[i][j];
50  }
51  delete[] matrix[i];
52  }
53  delete[] matrix;
54  }
55  //-------------------------------------------------------------------------------------------------
56  // Allocate 2d array
57  //-------------------------------------------------------------------------------------------------
58  template<class T>
59  void Allocate2dArray(T** &matrix, Vec2u res)
60  {
61  matrix = new T*[res[0]];
62  for (size_t i = 0; i < res[0]; i++) {
63  matrix[i] = new T[res[1]];
64  }
65  }
66  template<class T>
67  void Delete2dArray(T** matrix, Vec2u res)
68  {
69  for (size_t i = 0; i < res[0]; i++) {
70  delete[] matrix[i];
71  }
72  delete[] matrix;
73  }

83  template<class T>

89 }
