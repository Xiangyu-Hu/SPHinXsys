227  {
228  Real operator()(Real input, Real production_rate, Real loss_rate, Real dt) const
229  {
230  return input * exp(-loss_rate * dt) + production_rate * (1.0 - exp(-loss_rate * dt)) / (loss_rate + TinyReal);
