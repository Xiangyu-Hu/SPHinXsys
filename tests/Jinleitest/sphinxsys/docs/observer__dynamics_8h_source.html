73  virtual void Interaction(size_t index_i, Real dt = 0.0) override
74  {
75  VariableType observed_quantity(0);
76  Real ttl_weight(0);
77 
78  for (size_t k = 0; k < this->contact_configuration_.size(); ++k)
79  {
80  StdLargeVec<Real> &Vol_k = *(contact_Vol_[k]);
81  StdLargeVec<VariableType> &data_k = *(contact_data_[k]);
82  Neighborhood &contact_neighborhood = (*this->contact_configuration_[k])[index_i];
83  for (size_t n = 0; n != contact_neighborhood.current_size_; ++n)
84  {
85  size_t index_j = contact_neighborhood.j_[n];
86  Real weight_j = contact_neighborhood.W_ij_[n] * Vol_k[index_j];
87 
88  observed_quantity += weight_j * data_k[index_j];
89  ttl_weight += weight_j;
90  }
91  }
92  (*interpolated_quantities_)[index_i] = observed_quantity / (ttl_weight + TinyReal);
93  };
94  };
