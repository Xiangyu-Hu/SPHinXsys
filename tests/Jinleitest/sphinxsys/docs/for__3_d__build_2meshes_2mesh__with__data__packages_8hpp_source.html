18  DataType GridDataPackage<PKG_SIZE, ADDRS_SIZE>::probeDataPackage(PackageDataAddress<DataType> &pkg_data_addrs, const Vecd &position)
19  {
20  Vec3u grid_idx = CellIndexFromPosition(position);
21  Vec3d grid_pos = GridPositionFromIndex(grid_idx);
22  Vec3d alpha = (position - grid_pos) / grid_spacing_;
23  Vec3d beta = Vec3d(1.0) - alpha;
24 
25  DataType bilinear_1 = *pkg_data_addrs[grid_idx[0]][grid_idx[1]][grid_idx[2]] * beta[0] * beta[1] +
26  *pkg_data_addrs[grid_idx[0] + 1][grid_idx[1]][grid_idx[2]] * alpha[0] * beta[1] +
27  *pkg_data_addrs[grid_idx[0]][grid_idx[1] + 1][grid_idx[2]] * beta[0] * alpha[1] +
28  *pkg_data_addrs[grid_idx[0] + 1][grid_idx[1] + 1][grid_idx[2]] * alpha[0] * alpha[1];
29  DataType bilinear_2 = *pkg_data_addrs[grid_idx[0]][grid_idx[1]][grid_idx[2] + 1] * beta[0] * beta[1] +
30  *pkg_data_addrs[grid_idx[0] + 1][grid_idx[1]][grid_idx[2] + 1] * alpha[0] * beta[1] +
31  *pkg_data_addrs[grid_idx[0]][grid_idx[1] + 1][grid_idx[2] + 1] * beta[0] * alpha[1] +
32  *pkg_data_addrs[grid_idx[0] + 1][grid_idx[1] + 1][grid_idx[2] + 1] * alpha[0] * alpha[1];
33  return bilinear_1 * beta[2] + bilinear_2 * alpha[2];
34  }

37  template <typename InDataType, typename OutDataType>
38  void GridDataPackage<PKG_SIZE, ADDRS_SIZE>::
39  computeGradient(PackageDataAddress<InDataType> &in_pkg_data_addrs,
40  PackageDataAddress<OutDataType> out_pkg_data_addrs, Real dt)
41  {
42  for (int i = 1; i != PKG_SIZE + 1; ++i)
43  for (int j = 1; j != PKG_SIZE + 1; ++j)
44  for (int k = 1; k != PKG_SIZE + 1; ++k)
45  {
46  Real dphidx = (*in_pkg_data_addrs[i + 1][j][k] - *in_pkg_data_addrs[i - 1][j][k]);
47  Real dphidy = (*in_pkg_data_addrs[i][j + 1][k] - *in_pkg_data_addrs[i][j - 1][k]);
48  Real dphidz = (*in_pkg_data_addrs[i][j][k + 1] - *in_pkg_data_addrs[i][j][k - 1]);
49  *out_pkg_data_addrs[i][j][k] = 0.5 * Vecd(dphidx, dphidy, dphidz) / grid_spacing_;
50  }
51  }
