450  class DampingBySplittingAlgorithm :
451  public InteractionDynamicsSplitting,
452  public DataDelegateInner<SPHBody, BaseParticles, BaseMaterial>
453  {
454  public:
455  DampingBySplittingAlgorithm(SPHBodyInnerRelation* body_inner_relation,
456  StdLargeVec<VariableType>& variable, Real eta) :
457  InteractionDynamicsSplitting(body_inner_relation->sph_body_),
458  DataDelegateInner<SPHBody, BaseParticles, BaseMaterial>(body_inner_relation),
459  Vol_(particles_->Vol_), mass_(particles_->mass_), variable_(variable), eta_(eta) {};
460  virtual ~DampingBySplittingAlgorithm() {};
461  protected:
462  StdLargeVec<Real>& Vol_, & mass_;
463  StdLargeVec<VariableType>& variable_;
464  Real eta_;
465  virtual void Interaction(size_t index_i, Real dt = 0.0) override
466  {
467  Real Vol_i = Vol_[index_i];
468  Real mass_i = mass_[index_i];
469  VariableType& variable_i = variable_[index_i];
470 
471  VariableType error(0);
472  Real parameter_a(0);
473  Real parameter_c(0);
474  Neighborhood& inner_neighborhood = inner_configuration_[index_i];
475  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
476  {
477  size_t index_j = inner_neighborhood.j_[n];
478  //linear projection
479  VariableType variable_derivative = (variable_i - variable_[index_j]);
480  Real parameter_b = 2.0 * eta_ * inner_neighborhood.dW_ij_[n]
481  * Vol_i * Vol_[index_j] * dt / inner_neighborhood.r_ij_[n];
482 
483  error -= variable_derivative * parameter_b;
484  parameter_a += parameter_b;
485  parameter_c += parameter_b * parameter_b;
486  }
487 
488  parameter_a -= mass_i;
489  Real parameter_l = parameter_a * parameter_a + parameter_c;
490  VariableType parameter_k = error / (parameter_l + TinyReal);
491  variable_[index_i] += parameter_k * parameter_a;
492 
493  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
494  {
495  size_t index_j = inner_neighborhood.j_[n];
496 
497  Real parameter_b = 2.0 * eta_ * inner_neighborhood.dW_ij_[n]
498  * Vol_i * Vol_[index_j] * dt / inner_neighborhood.r_ij_[n];
499 
500  //predicted quantity at particle j
501  VariableType variable_j = variable_[index_j] - parameter_k * parameter_b;
502  VariableType variable_derivative = (variable_i - variable_j);
503 
504  //exchange in conservation form
505  variable_[index_j] -= variable_derivative * parameter_b / mass_[index_j];
506  }
507  };
508  };

519  class DampingBySplittingPairwise :
520  public InteractionDynamicsSplitting,
521  public DataDelegateInner<SPHBody, BaseParticles, BaseMaterial>
522  {
523  public:
524  DampingBySplittingPairwise(SPHBodyInnerRelation* body_inner_relation,
525  StdLargeVec<VariableType>& variable, Real eta) :
526  InteractionDynamicsSplitting(body_inner_relation->sph_body_),
527  DataDelegateInner<SPHBody, BaseParticles, BaseMaterial>(body_inner_relation),
528  Vol_(particles_->Vol_), mass_(particles_->mass_), variable_(variable), eta_(eta) {};
529  virtual ~DampingBySplittingPairwise() {};
530  protected:
531  StdLargeVec<Real>& Vol_, & mass_;
532  StdLargeVec<VariableType>& variable_;
533  Real eta_;
535  virtual void Interaction(size_t index_i, Real dt = 0.0) override
536  {
537  Real Vol_i = Vol_[index_i];
538  Real mass_i = mass_[index_i];
539  VariableType& variable_i = variable_[index_i];
540 
541  StdVec<Real> parameter_b(MaximumNeighborhoodSize);
542  Neighborhood& inner_neighborhood = inner_configuration_[index_i];
543  //forward sweep
544  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
545  {
546  size_t index_j = inner_neighborhood.j_[n];
547  Real mass_j = mass_[index_j];
548 
549  VariableType variable_derivative = (variable_i - variable_[index_j]);
550  parameter_b[n] = eta_ * inner_neighborhood.dW_ij_[n]
551  * Vol_i * Vol_[index_j] * dt / inner_neighborhood.r_ij_[n];
552 
553  VariableType increment = parameter_b[n] * variable_derivative
554  / (mass_i * mass_j - parameter_b[n] * (mass_i + mass_j));
555  variable_[index_i] += increment * mass_j;
556  variable_[index_j] -= increment * mass_i;
557  }
558 
559  //backward sweep
560  for (size_t n = inner_neighborhood.current_size_; n != 0; --n)
561  {
562  size_t index_j = inner_neighborhood.j_[n - 1];
563  Real mass_j = mass_[index_j];
564 
565  VariableType variable_derivative = (variable_i - variable_[index_j]);
566  VariableType increment = parameter_b[n - 1] * variable_derivative
567  / (mass_i * mass_j - parameter_b[n - 1] * (mass_i + mass_j));
568 
569  variable_[index_i] += increment * mass_j;
570  variable_[index_j] -= increment * mass_i;
571  }
572  };
