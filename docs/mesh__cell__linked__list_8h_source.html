49  inline size_t median_of_three(const RandomAccessIterator& array, size_t l, size_t m, size_t r) const {
50  return comp_(array[l], array[m]) ? (comp_(array[m], array[r]) ? m : (comp_(array[l], array[r]) ? r : l))
51  : (comp_(array[r], array[m]) ? m : (comp_(array[r], array[l]) ? r : l));
52  }
53 
54  inline size_t PseudoMedianOfNine(const RandomAccessIterator& array, const QuickSortParticleRange& range) const {
55  size_t offset = range.size_ / 8u;
56  return median_of_three(array,
57  median_of_three(array, 0, offset, offset * 2),
58  median_of_three(array, offset * 3, offset * 4, offset * 5),
59  median_of_three(array, offset * 6, offset * 7, range.size_ - 1));
60 
61  }

63  size_t splitRange(QuickSortParticleRange& range) {
64  RandomAccessIterator array = range.begin_;
65  RandomAccessIterator key0 = range.begin_;
66  size_t m = PseudoMedianOfNine(array, range);
67  if (m) swap_particle_data_(array, array + m);
68 
69  size_t i = 0;
70  size_t j = range.size_;
71  // Partition interval [i+1,j-1] with key *key0.
72  for (;;) {
73  __TBB_ASSERT(i < j, NULL);
74  // Loop must terminate since array[l]==*key0.
75  do {
76  --j;
77  __TBB_ASSERT(i <= j, "bad ordering relation?");
78  } while (comp_(*key0, array[j]));
79  do {
80  __TBB_ASSERT(i <= j, NULL);
81  if (i == j) goto quick_sort_particle_partition;
82  ++i;
83  } while (comp_(array[i], *key0));
84  if (i == j) goto quick_sort_particle_partition;
85  swap_particle_data_(array + i, array + j);
86  }
87  quick_sort_particle_partition:
88  // Put the partition key were it belongs
89  swap_particle_data_(array + j, key0);
90  // array[l..j) is less or equal to key.
91  // array(j..r) is greater or equal to key.
92  // array[j] is equal to key
93  i = j + 1;
94  size_t new_range_size = range.size_ - i;
95  range.size_ = j;
96  return new_range_size;
97  }

123  /*
124  Description : QuickSort in Iterator format
125  Link : https://stackoverflow.com/a/54976413/3547485
126  Ref : http://www.cs.fsu.edu/~lacher/courses/COP4531/lectures/sorts/slide09.html
127  */
128  template <typename RandomAccessIterator, typename Compare, typename SwapType>
129  RandomAccessIterator Partition(RandomAccessIterator first, RandomAccessIterator last, Compare& compare, SwapType& swap_particle_data)
130  {
131  auto pivot = std::prev(last, 1);
132  auto i = first;
133  for (auto j = first; j != pivot; ++j) {
134  // bool format
135  if (compare(*j, *pivot)) {
136  swap_particle_data(i++, j);
137  }
138  }
139  swap_particle_data(i, pivot);
140  return i;
141  }

143  template <typename RandomAccessIterator, typename Compare, typename SwapType>
144  void SerialQuickSort(RandomAccessIterator first, RandomAccessIterator last, Compare& compare, SwapType& swap_particle_data)
145  {
146  if (std::distance(first, last) > 1) {
147  RandomAccessIterator bound = Partition(first, last, compare, swap_particle_data);
148  SerialQuickSort(first, bound, compare, swap_particle_data);
149  SerialQuickSort(bound + 1, last, compare, swap_particle_data);
150  }
151  }

158  template< typename RandomAccessIterator, typename Compare, typename SwapType>
159  void InsertionSort(RandomAccessIterator First, RandomAccessIterator Last, Compare& compare, SwapType& swap_particle_data)
160  {
161  RandomAccessIterator min = First;
162  for (RandomAccessIterator i = First + 1; i < Last; ++i)
163  if (compare(*i, *min)) min = i;
164 
165  swap_particle_data(First, min);
166  while (++First < Last)
167  for (RandomAccessIterator j = First; compare(*j, *(j - 1)); --j)
168  swap_particle_data((j - 1), j);
169  }
