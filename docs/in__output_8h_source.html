266  std::ofstream out_file(filefullpath_.c_str(), std::ios::app);
267  out_file << "run_time" << " ";
268  for (size_t i = 0; i != base_particles_->total_real_particles_; ++i)
269  {
270  std::string quantity_name_i = quantity_name + "[" + std::to_string(i) + "]";
271  plt_engine_.writeAQuantityHeader(out_file, this->interpolated_quantities_[i], quantity_name_i);
272  }
