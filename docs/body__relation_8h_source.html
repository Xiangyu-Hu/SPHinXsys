59  struct SearchRangeMultiResolution
60  {
61  int search_range_;
62  SearchRangeMultiResolution(SPHBody* body, SPHBody* contact_body) : search_range_(0)
63  {
64  int body_refinement_level = body->particle_adaptation_->GlobalRefinementLevel();
65  int contact_body_refinement_level = contact_body->particle_adaptation_->GlobalRefinementLevel();
66  search_range_ = body_refinement_level >= contact_body_refinement_level ?
67  1 : powerN(2, contact_body_refinement_level - body_refinement_level);
68  };
69  int operator () (size_t particle_index) const { return search_range_; };
70  };

73  struct SearchRangeVariableSmoothingLength

78  SearchRangeVariableSmoothingLength(SPHBody* body, MeshCellLinkedList* target_mesh_cell_linked_list) :
